Simplify Directory Path

Given a string A representing an absolute path for a file (Unix-style).

Return the string A after simplifying the absolute path.

Note:

  1. Absolute path always begin with ’/’ ( root directory ).

  2. Path will not have whitespace characters.

import java.util.Stack;

public class SimplyfiyDirectory {
    public String simplifyPath(String A) {
      //  if(A.compareTo("/") == 0) return "/";
        Stack s = new Stack<>();
        String[] paths = A.split("/");
        for(int i = 0 ; i < paths.length; i++) {
            if(paths[i].length() != 0) {
                if (paths[i].compareTo(".") == 0) continue;
                else if (paths[i].compareTo("..") == 0){ if(!s.empty())s.pop(); }
                else s.push(paths[i]);
            }
        }
        StringBuilder res = new StringBuilder();
        while (!s.empty()) {
            res.insert(0,s.pop());
            res.insert(0,"/");
        }
        return res.length() == 0 ? "/" : res.toString();
    }
}
