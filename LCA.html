public class LCA {

    public TreeNode lcaHelper(TreeNode curr, int A, int B) {
        if(curr == null) {
            return null;
        }
        boolean t1 = curr.val == A || curr.val == B;
        TreeNode left = lcaHelper(curr.left, A, B);
        TreeNode right = lcaHelper(curr.right, A, B);
        if(t1) return curr;
        if(left != null && right != null) return curr;
        else if(left == null && right == null) return null;
        else if(left != null) return left;
        else return right;
    }

    public int lca(TreeNode A, int B, int C) {
        TreeNode temp = lcaHelper(A, B, C);
        return temp.val;
    }
}
